29 //struct to use for a pair of two reco particles, to make sure the correct ones stay together
30 struct RecoParticlePair {
31 edm4hep::ReconstructedParticleData particle_1;
32 edm4hep::ReconstructedParticleData particle_2;
33 TLorentzVector merged_TLV(){
34 TLorentzVector tlv_1 = getTLV_reco(particle_1);
35 TLorentzVector tlv_2 = getTLV_reco(particle_2);
36 return tlv_1+tlv_2;
37 }
38 void sort_by_pT(){
39 double pT_1 = sqrt(particle_1.momentum.x * particle_1.momentum.x + particle_1.momentum.y * particle_1.momentum.y);
40 double pT_2 = sqrt(particle_2.momentum.x * particle_2.momentum.x + particle_2.momentum.y * particle_2.momentum.y);
41
42 if (pT_1 >= pT_2){ return;} //nothing to do if already sorted corrected
43 else {
44 edm4hep::ReconstructedParticleData sublead = particle_1;
45
46 particle_1 = particle_2;
47 particle_2 = sublead;
48 return;
49
50 }
51
52 }
53 } ;
